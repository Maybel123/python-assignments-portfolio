#%% 
import numpy as np
import matplotlib.pyplot as plt

def gradient_descent(x0, y0,f, grad_f, alpha, num_iterations):
    x, y = x0, y0  
    for i in range(num_iterations):
        grad_x, grad_y = grad_f(x,y)
        
        x = x-alpha*grad_x 
        y = y-alpha*grad_y 

        if i % (num_iterations // 10) == 0:
            print(f"Interation {i}: x={x:.4f}, y={y:.4f}, f(x,y)={f(x,y):.4f}")

    return x, y 
#%% 
def fun_1(x,y):
    return x**2 + y**2

def grad_f_1(x,y):
    grad_x = 2*x 
    grad_y = 2*y 
    return grad_x, grad_y

x_min, y_min = gradient_descent(0.1, 0.1, fun_1, grad_f_1, 0.1, 10)
print(f"Final result for (0.1, 0.1): x = {x_min}, y = {y_min}")

x_min, y_min = gradient_descent(-1, 1, fun_1, grad_f_1, 0.01, 100)
print(f"Final result for (-1, 1): x = {x_min}, y = {y_min}")
#%% 
def fun_2(x,y):
    return 1-np.exp(-x**2-(y-2)**2)-2*np.exp(-x**2-(y+2)**2)

def grad_f_2(x,y):
    grad_x = 2*x*np.exp(-x**2 - (y-2)**2) + 4*x*np.exp(-x**2 - (y+2)**2) 
    grad_y = -(4 - 2*y)*np.exp(-x**2 - (y-2)**2) - 2*(-2*y - 4)*np.exp(-x**2 - (y+2)**2)
    return grad_x, grad_y

x_min, y_min = gradient_descent(0, 1, fun_2, grad_f_2, 0.01, 10000)
print(f"Final result for (0, 1): x = {x_min}, y = {y_min}")

x_min, y_min = gradient_descent(0, -1, fun_2, grad_f_2, 0.01, 10000)
print(f"Final result for (0, -1): x = {x_min}, y = {y_min}")
#%% 
X=np.linspace(-5,5,100)
Y=np.linspace(-5,5,100)
x,y=np.meshgrid(X,Y)
z= fun_2(x,y)

fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')
ax.plot_surface(x, y, z,cmap='viridis', edgecolor='none') 

ax.set_xlabel('X')

ax.set_title('Plot of f2(x,y)')

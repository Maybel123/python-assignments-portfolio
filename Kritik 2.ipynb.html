import numpy as np

def roots(f, a, b, tol=1e-10):
    if f(a) * f(b) >= 0:
        return None

    for i in range(1000): 
        mid = (a + b) / 2
        if abs(f(mid)) < tol: 
            return mid
        elif f(a) * f(mid) < 0:
            b = mid
        else:
            a = mid
